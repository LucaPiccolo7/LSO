3) Si realizzi uno script "scriviNumeri.sh" che scrive a video i numeri da 0 ad N: 0,1,2,...,N-1
	Il valore di N viene passato allo script tramite riga di comando

#!/bin/bash
if [ $# -eq 1 ]
then
	echo "Argomento $1 accettato"

	i=0
	while [ $i -lt $1 ]
	do
		echo $i
		i=$(( $i + 1 ))
	done

	exit 0
else
	echo "Non Ã¨ stato passato esattamente un numero come argomento"
	exit -1
fi

4) Si realizzi uno script che chiameremo "scriviNumeri.sh" che
	scrive a video i numeri da 20 a 10: 20,19,18,...,10

#!/bin/bash
i=20
while [ $i -le 10 ]
do
	echo "$i"
	i=$(( $i - 1 ))
